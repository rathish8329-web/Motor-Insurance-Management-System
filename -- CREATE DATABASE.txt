-- CREATE DATABASE

CREATE DATABASE MOTOR_insurance;
USE MOTOR_insurance;

-- create table MI_regions
 
CREATE TABLE MI_regions (
    Region_id INT AUTO_INCREMENT PRIMARY KEY,
    Region_name VARCHAR(200) NOT NULL UNIQUE,
    dateofbrith date,
    phone varbinary(100),
    Status VARCHAR(30) DEFAULT 'ACTIVE',
    Created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- create table MI_states

CREATE TABLE MI_states (
    State_id INT AUTO_INCREMENT PRIMARY KEY,
    Region_id INT NOT NULL,
    State_name VARCHAR(200) NOT NULL,
    Status VARCHAR(30) DEFAULT 'ACTIVE',
    FOREIGN KEY (region_id) REFERENCES mi_regions(region_id)
);

CREATE TABLE MI_cities (
    city_id INT AUTO_INCREMENT PRIMARY KEY,
    state_id INT NOT NULL,
    city_name VARCHAR(200) NOT NULL,
    Pincode VARCHAR(30),
    Address varchar(300),
    
    Status VARCHAR(20) DEFAULT 'ACTIVE',
    FOREIGN KEY (state_id) REFERENCES mi_states(state_id)
);


CREATE TABLE MI_branches (
    Branch_id INT AUTO_INCREMENT PRIMARY KEY,
    City_id INT NOT NULL,
    Branch_name VARCHAR(200) NOT NULL,
    pincode varchar(50),
    Branch_code VARCHAR(30) UNIQUE,
    Address VARCHAR(255),
    Status VARCHAR(20) DEFAULT 'ACTIVE',
    FOREIGN KEY (city_id) REFERENCES mi_cities(city_id)
);


CREATE TABLE MI_roles (
    Role_id INT AUTO_INCREMENT PRIMARY KEY,
    Role_name VARCHAR(50) UNIQUE NOT NULL,
    Role_type varchar(30),
    Status VARCHAR(20) DEFAULT 'ACTIVE'
);

-- create table MI_USER
CREATE TABLE MI_users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    branch_id INT NOT NULL,
    role_id INT NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(100) NOT NULL,
    full_name VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    mobile VARCHAR(15),
    status VARCHAR(20) DEFAULT 'ACTIVE',
    FOREIGN KEY (branch_id) REFERENCES mi_branches(branch_id),
    FOREIGN KEY (role_id) REFERENCES mi_roles(role_id)
);


CREATE TABLE MI_CUSTOMERS (
    customer_id INT AUTO_INCREMENT PRIMARY KEY,
    customer_name VARCHAR(100) NOT NULL,
    gender VARCHAR(10),
    dob DATE,
    mobile VARCHAR(15) UNIQUE,
    email VARCHAR(209) UNIQUE,
    address VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE mi_vehicles (
    vehicle_id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT NOT NULL,
    registration_no VARCHAR(20) UNIQUE NOT NULL,
    make VARCHAR(50),
    model VARCHAR(50),
    manufacturing_year YEAR,
    engine_no VARCHAR(50) UNIQUE,
    chassis_no VARCHAR(50) UNIQUE,
    FOREIGN KEY (customer_id) REFERENCES mi_customers(customer_id)
);


CREATE TABLE mi_quotes (
    quote_id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT NOT NULL,
    vehicle_id INT NOT NULL,
    quote_date DATE,
    premium_amount DECIMAL(12,2),
    quote_status VARCHAR(20) DEFAULT 'PENDING',
    FOREIGN KEY (customer_id) REFERENCES mi_customers(customer_id),
    FOREIGN KEY (vehicle_id) REFERENCES mi_vehicles(vehicle_id)
);

CREATE TABLE mi_policies (
    policy_id INT AUTO_INCREMENT PRIMARY KEY,
    quote_id INT NOT NULL,
    policy_number VARCHAR(50) UNIQUE NOT NULL,
    issue_date DATE,
    start_date DATE,
    end_date DATE,
    premium_amount DECIMAL(12,2),
    sum_insured DECIMAL(15,2),
    policy_status VARCHAR(20) DEFAULT 'ACTIVE',
    FOREIGN KEY (quote_id) REFERENCES mi_quotes(quote_id)
);

CREATE TABLE mi_payments (
    payment_id INT AUTO_INCREMENT PRIMARY KEY,
    policy_id INT NOT NULL,
    payment_date DATE,
    amount DECIMAL(12,2),
    payment_mode VARCHAR(30),
    transaction_no VARCHAR(100) UNIQUE,
    FOREIGN KEY (policy_id) REFERENCES mi_policies(policy_id)
);


CREATE TABLE mi_claims (
    claim_id INT AUTO_INCREMENT PRIMARY KEY,
    policy_id INT NOT NULL,
    claim_date DATE,
    claim_amount DECIMAL(12,2),
    approved_amount DECIMAL(12,2),
    claim_status VARCHAR(20) DEFAULT 'PENDING',
    FOREIGN KEY (policy_id) REFERENCES mi_policies(policy_id)
);


CREATE TABLE mi_surveyors (
    surveyor_id INT AUTO_INCREMENT PRIMARY KEY,
    surveyor_name VARCHAR(10) NOT NULL,
    mobile VARCHAR(15),
    email VARCHAR(100),
    license_no VARCHAR(50) UNIQUE
);


CREATE TABLE mi_renewals (
    renewal_id INT AUTO_INCREMENT PRIMARY KEY,
    policy_id INT NOT NULL,
    renewal_date DATE,
    new_expiry_date DATE,
    renewal_status VARCHAR(20) DEFAULT 'PENDING',
    FOREIGN KEY (policy_id) REFERENCES mi_policies(policy_id)
);


CREATE TABLE mi_audit_logs (
    log_id INT AUTO_INCREMENT PRIMARY KEY,
    table_name VARCHAR(100),
    action_type VARCHAR(20),
    action_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    user_name VARCHAR(100),
    remarks VARCHAR(255)
);
